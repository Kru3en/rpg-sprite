export class SpriteRenderer {
    constructor(previewCanvas, spritesheetCanvas) {
        this.previewCtx = previewCanvas.getContext('2d');
        this.spritesheetCtx = spritesheetCanvas.getContext('2d');
        this.spritesheetCanvas = spritesheetCanvas;
        this.previewCtx.imageSmoothingEnabled = false;
        this.spritesheetCtx.imageSmoothingEnabled = false;
        this.layers = [];
    }

    addLayer(image, zIndex) {
        if (image) this.layers.push({ image, zIndex });
        this.layers.sort((a, b) => a.zIndex - b.zIndex);
    }

    clearLayers() {
        this.layers = [];
    }

    renderPreview(frame, direction) {
        this.previewCtx.clearRect(0, 0, 128, 128);
        const x = 32, y = 32;

        this.layers.forEach(layer => {
            if (layer.image) {
                this.previewCtx.drawImage(
                    layer.image,
                    frame * 64, direction * 64, 64, 64,
                    x, y, 64, 64
                );
            }
        });
    }

    renderSpritesheet(frames, directions) {
        this.spritesheetCtx.clearRect(0, 0, frames * 64, directions * 64);
        this.spritesheetCanvas.width = frames * 64;
        this.spritesheetCanvas.height = directions * 64;

        for (let dir = 0; dir < directions; dir++) {
            for (let frame = 0; frame < frames; frame++) {
                this.layers.forEach(layer => {
                    if (layer.image) {
                        this.spritesheetCtx.drawImage(
                            layer.image,
                            frame * 64, dir * 64, 64, 64,
                            frame * 64, dir * 64, 64, 64
                        );
                    }
                });
            }
        }
    }

    exportSpritesheet() {
        return this.spritesheetCanvas.toDataURL('image/png');
    }

}export class AssetLoader {
    constructor() {
        this.basePath = './assets/sprite/';
        this.cache = new Map();
    }

    async loadSprite(path) {
        if (this.cache.has(path)) return this.cache.get(path);

        const img = new Image();
        return new Promise((resolve, reject) => {
            img.onload = () => {
                this.cache.set(path, img);
                resolve(img);
            };
            img.onerror = () => {
                console.warn(`Failed to load image: ${path}`);
                resolve(null);
            };
            img.src = this.basePath + path;
        });
    }

    getSpritePath(part, config) {
        const { gender, bodyType, skinColor, earStyle, eyeColor, expression, hairStyle, hairColor, 
                facialHair, facialHairColor, torso, torsoColor, torso2, torso2Color, torso3, torso3Color, 
                legs, legsColor, feet, feetColor, weaponType, weapon, animation } = config;

        // Карта базовых типов и их производных
        const genderMap = {
            male: bodyType === 'muscular' ? 'Human_male_muscular' : 'Human_male',
            female: bodyType === 'pregnant' ? 'Human_female_pregnant' : 'Human_female',
            androgynous: 'Human_androgynous',
            child: 'Human_child',
            orc: 'Orc_male',
            skeleton: 'Skeleton'
        };
        const genderFolder = genderMap[gender] || 'Human_male';

        // Карта замен для производных типов
        const baseFolderMap = {
            'Human_male_muscular': 'Human_male',
            'Human_female_pregnant': 'Human_female',
            'Human_male': 'Human_male',
            'Human_female': 'Human_female',
            'Human_androgynous': 'Human_androgynous',
            'Human_child': 'Human_child',
            'Orc_male': 'Orc_male',
            'Skeleton': 'Skeleton'
        };
        const baseFolder = baseFolderMap[genderFolder] || genderFolder;

        const defaultSkin = ['Orc_male', 'Skeleton'].includes(genderFolder) ? '' : skinColor;

        const paths = {
            body: () => defaultSkin ? 
                `character/Body/Base/${genderFolder}/${defaultSkin}/${animation}.png` : 
                `character/Body/Base/${genderFolder}/${animation}.png`,
            ears: () => {
                if (['Orc_male', 'Skeleton', 'Human_androgynous'].includes(genderFolder) || earStyle === 'none') return null;
                return genderFolder === baseFolder ?
                    `character/Body/Ears/${genderFolder}/${earStyle}/${defaultSkin}/${animation}.png` :
                    `character/Body/Ears/${baseFolder}/${earStyle}/${defaultSkin}/${animation}.png`;
            },
            eyes: () => {
                if (!eyeColor || !['Human_child', 'Human_female', 'Human_male', 'Human_male_muscular', 'Human_female_pregnant'].includes(genderFolder)) return null;
                return genderFolder === baseFolder ?
                    `character/Body/Eyes/${genderFolder}/${eyeColor}/${animation}.png` :
                    `character/Body/Eyes/${baseFolder}/${eyeColor}/${animation}.png`;
            },
            expression: () => {
                if (expression === 'none' || !['Human_female', 'Human_male', 'Human_male_muscular', 'Human_female_pregnant'].includes(genderFolder)) return null;
                return genderFolder === baseFolder ?
                    `character/Expression/${genderFolder}/${expression}/${defaultSkin}/${animation}.png` :
                    `character/Expression/${baseFolder}/${expression}/${defaultSkin}/${animation}.png`;
            },
            hair: () => {
                if (hairStyle === 'none' || ['Skeleton'].includes(genderFolder)) return null;
                return genderFolder === baseFolder ?
                    `character/Body/Hair/${genderFolder}/${hairStyle}/${hairColor}/${animation}.png` :
                    `character/Body/Hair/${baseFolder}/${hairStyle}/${hairColor}/${animation}.png`;
            },
            facialHair: () => {
                if (facialHair === 'none' || !['Human_male', 'Human_male_muscular'].includes(genderFolder)) return null;
                return genderFolder === baseFolder ?
                    `character/Body/Facial Hair/${genderFolder}/${facialHair}/${facialHairColor}/${animation}.png` :
                    `character/Body/Facial Hair/${baseFolder}/${facialHair}/${facialHairColor}/${animation}.png`;
            },
            torso: () => torso !== 'none' && torsoColor ? 
                `character/Body/Torso/${genderFolder}/${torso}/${torsoColor}/${animation}.png` : null,
            torso2: () => torso2 !== 'none' && torso2Color ? 
                `character/Body/Torso 2/${genderFolder}/${torso2}/${torso2Color}/${animation}.png` : null,
            torso3: () => torso3 !== 'none' && torso3Color ?

            weapon: () => weapon !== 'none' && weaponType !== 'none' ? 
                `character/Weapons/${genderFolder}/${weaponType}/${weapon}/${animation}.png` : null
        };

        const path = paths[part] ? paths[part]() : null;

}export class AnimationController {
    constructor(spriteRenderer) {
        this.spriteRenderer = spriteRenderer;
        this.currentAnimation = 'idle';
        this.frame = 0;
        this.direction = 0;
        this.isPlaying = false;
        this.speed = 10;
        this.animations = {
            idle: { frames: 4, directions: 4, loop: true },
            walk: { frames: 8, directions: 4, loop: true },
            swing: { frames: 6, directions: 4, loop: false },
            thrust: { frames: 8, directions: 4, loop: false },
            shoot: { frames: 13, directions: 4, loop: false },
            magic: { frames: 7, directions: 4, loop: false },
            hurt: { frames: 6, directions: 1, loop: false },
            run: { frames: 6, directions: 4, loop: true } // Добавлено для Human_male
        };
    }

    setAnimation(type) {
        if (this.animations[type]) {
            this.currentAnimation = type;
            this.frame = 0;
            this.spriteRenderer.renderSpritesheet(
                this.animations[type].frames,
                this.animations[type].directions
            );
            this.renderFrame();
        }
    }

    setDirection(direction) {
        this.direction = Math.min(Math.max(direction, 0), this.animations[this.currentAnimation].directions - 1);
        this.renderFrame();
    }

    setSpeed(speed) {
        this.speed = speed;
    }

    play() {
        if (!this.isPlaying) {
            this.isPlaying = true;
            this.animate();
        }
    }

    stop() {
        this.isPlaying = false;
    }

    animate() {
        if (!this.isPlaying) return;

        this.renderFrame();
        const anim = this.animations[this.currentAnimation];
        this.frame++;

        if (this.frame >= anim.frames) {
            this.frame = anim.loop ? 0 : anim.frames - 1;
            if (!anim.loop) this.stop();
        }

        setTimeout(() => this.animate(), 1000 / this.speed);
    }

    renderFrame() {
        this.spriteRenderer.renderPreview(this.frame, this.direction);
    }
}
